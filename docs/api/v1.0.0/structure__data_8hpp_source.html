198 // check optional date string
199 // -> either default or "YYYY-MM-DD" (only string format checked, not date)
200 bool isValidDateFormatOptional(const std::string& s) {
201  // default?
202  if (isDefaultValue(s)) return true;
203  // check length
204  if (s.length() != 10) return false;
205  // check delimiters
206  if (s[4] != '-' || s[7] != '-') return false;
207  // check format
208  std::istringstream is(s);
209  int d, m, y;
210  char dash1, dash2;
211  if (is >> y >> dash1 >> m >> dash2 >> d) {
212  return (dash1 == '-' && dash2 == '-');
213  } else {
214  return false;
215  }
216 }

218 // check if optional vector has right size
219 template<typename T>
220 bool hasRightSizeOptional(const std::vector<T>& v, int exp_size) {
221  return (isDefaultValue(v) || v.size() == exp_size);
222 }

224 // check if all indices in vector are in [0, num-1] (T = integer type)
225 template<typename T, typename Tnum>
226 bool hasValidIndices(const T* v, size_t size, Tnum num) {
227  T tnum = T(num);
228  for (size_t i = 0; i < size; ++i) {
229  if (v[i] < T(0) || v[i] >= tnum) return false;
230  }
231  return true;
232 }
233 template<typename T, typename Tnum>
234 bool hasValidIndices(const std::vector<T>& v, Tnum num) {
235  if (v.empty()) return true;
236  else return hasValidIndices(&v[0], v.size(), num);
237 }

273 bool StructureData::hasConsistentData() {
274  // check unitCell: if given, must be of length 6
275  if (!hasRightSizeOptional(unitCell, 6)) return false;
276  // check dates
277  if (!isValidDateFormatOptional(depositionDate)) return false;
278  if (!isValidDateFormatOptional(releaseDate)) return false;
279  // check ncsOperatorList: all elements must have length 16
280  for (size_t i = 0; i < ncsOperatorList.size(); ++i) {
281  if (ncsOperatorList[i].size() != 16) return false;
282  }
283  // check chain indices in bioAssembly-transforms and entities
284  for (size_t i = 0; i < bioAssemblyList.size(); ++i) {
285  const BioAssembly& ba = bioAssemblyList[i];
286  for (size_t j = 0; j < ba.transformList.size(); ++j) {
287  const Transform & t = ba.transformList[j];
288  if (!hasValidIndices(t.chainIndexList, numChains)) return false;
289  }
290  }
291  for (size_t i = 0; i < entityList.size(); ++i) {
292  const Entity& ent = entityList[i];
293  if (!hasValidIndices(ent.chainIndexList, numChains)) return false;
294  }
295  // check groups
296  for (size_t i = 0; i < groupList.size(); ++i) {
297  const GroupType& g = groupList[i];
298  const size_t num_atoms = g.formalChargeList.size();
299  if (g.atomNameList.size() != num_atoms) return false;
300  if (g.elementList.size() != num_atoms) return false;
301  if (g.bondAtomList.size() != g.bondOrderList.size() * 2) return false;
302  if (!hasValidIndices(g.bondAtomList, num_atoms)) return false;
303  }
304  // check global bonds
305  if (bondAtomList.size() != bondOrderList.size() * 2) return false;
306  if (!hasValidIndices(bondAtomList, numAtoms)) return false;
307  // check vector sizes
308  if (xCoordList.size() != numAtoms) return false;
309  if (yCoordList.size() != numAtoms) return false;
310  if (zCoordList.size() != numAtoms) return false;
311  if (!hasRightSizeOptional(bFactorList, numAtoms)) return false;
312  if (!hasRightSizeOptional(atomIdList, numAtoms)) return false;
313  if (!hasRightSizeOptional(altLocList, numAtoms)) return false;
314  if (!hasRightSizeOptional(occupancyList, numAtoms)) return false;
315  if (groupIdList.size() != numGroups) return false;
316  if (groupTypeList.size() != numGroups) return false;
317  if (!hasRightSizeOptional(secStructList, numGroups)) return false;
318  if (!hasRightSizeOptional(insCodeList, numGroups)) return false;
319  if (!hasRightSizeOptional(sequenceIndexList, numGroups)) return false;
320  if (chainIdList.size() != numChains) return false;
321  if (!hasRightSizeOptional(chainNameList, numChains)) return false;
322  if (groupsPerChain.size() != numChains) return false;
323  if (chainsPerModel.size() != numModels) return false;
324  // check indices
325  if (!hasValidIndices(groupTypeList, groupList.size())) return false;
326  // collect sequence lengths from entities and use to check
327  std::vector<int32_t> sequenceIndexSize(numChains);
328  for (size_t i = 0; i < entityList.size(); ++i) {
329  const Entity& ent = entityList[i];
330  for (size_t j = 0; j < ent.chainIndexList.size(); ++j) {
331  sequenceIndexSize[ent.chainIndexList[j]] = ent.sequence.length();
332  }
333  }
334  // traverse structure for more checks
335  int bond_count = bondOrderList.size();
336  int chain_idx = 0; // will be count at end of loop
337  int group_idx = 0; // will be count at end of loop
338  int atom_idx = 0; // will be count at end of loop
339  // traverse models
340  for (int model_idx = 0; model_idx < numModels; ++model_idx) {
341  // traverse chains
342  for (int j = 0; j < chainsPerModel[model_idx]; ++j, ++chain_idx) {
343  // check chain names (fixed length)
344  if (chainIdList[chain_idx].size() != 4) return false;
345  if (chainNameList[chain_idx].size() != 4) return false;
346  // traverse groups
347  for (int k = 0; k < groupsPerChain[chain_idx]; ++k, ++group_idx) {
348  // check seq. idx
349  if (!isDefaultValue(sequenceIndexList)) {
350  const int32_t idx = sequenceIndexList[group_idx];
351  // -1 is ok here
352  if (idx < -1 || idx >= sequenceIndexSize[chain_idx]) {
353  return false;
354  }
355  }
356  // count atoms
357  const GroupType& group = groupList[groupTypeList[group_idx]];
358  atom_idx += group.atomNameList.size();
359  // count bonds
360  bond_count += group.bondOrderList.size();
361  }
362  }
363  }
364  // check sizes
365  if (bond_count != numBonds) return false;
366  if (chain_idx != numChains) return false;
367  if (group_idx != numGroups) return false;
368  if (atom_idx != numAtoms) return false;
369 
370  return true;
371 }
