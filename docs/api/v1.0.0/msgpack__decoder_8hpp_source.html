39 // custom specialization for chars stored as strings
40 template <>
41 struct convert<char> {
42  const msgpack::object& operator()(const msgpack::object& obj,
43  char& c) const {
44  // extract string
45  std::string temp;
46  obj.convert(temp);
47  // ensure length
48  if (temp.size() != 1) {
49  throw mmtf::DecodeError("Observed single letter string not being "
50  "of length one!");
51  }
52  c = temp[0];
53  return obj;
54  }
55 };
