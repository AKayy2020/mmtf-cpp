101 MapDecoder::MapDecoder(const msgpack::object& obj) {
102  // sanity checks
103  if (obj.type != msgpack::type::MAP) {
104  throw DecodeError("Expected msgpack type to be MAP");
105  }
106  // get data
107  msgpack::object_kv* current_key_value = obj.via.map.ptr;
108  msgpack::object_kv* last_key_value = current_key_value + obj.via.map.size;
109  for (; current_key_value != last_key_value; ++current_key_value) {
110  msgpack::object* key = &(current_key_value->key);
111  msgpack::object* value = &(current_key_value->val);
112  if (key->type == msgpack::type::STR) {
113  std::string data_map_key(key->via.str.ptr, key->via.str.size);
114  data_map_[data_map_key] = value;
115  } else {
116  std::cerr << "Warning: Found non-string key type " << key->type
117  << "! Skipping..." << std::endl;
118  }
119  }
120 }

122 template<typename T>
123 void MapDecoder::decode(const std::string& key, bool required, T& target) {
124  // note: cost of O(M*log(N)) string comparisons (M parsed, N in map)
125  std::map<std::string, msgpack::object*>::iterator it;
126  it = data_map_.find(key);
127  if (it != data_map_.end()) {
128  checkType_(key, it->second->type, target);
129  if (it->second->type == msgpack::type::BIN) {
130  BinaryDecoder bd(*it->second, key);
131  bd.decode(target);
132  } else {
133  it->second->convert(target);
134  }
135  decoded_keys_.insert(key);
136  }
137  else if (required) {
138  throw DecodeError("MsgPack MAP does not contain required entry "
139  + key);
140  }
141 }

143 void MapDecoder::checkExtraKeys() {
144  // note: cost of O(N*log(M))) string comparisons (M parsed, N in map)
145  // simple set difference algorithm
146  std::map<std::string, msgpack::object*>::iterator map_it;
147  std::set<std::string>::iterator parsed_it;
148  for (map_it = data_map_.begin(); map_it != data_map_.end(); ++map_it) {
149  parsed_it = decoded_keys_.find(map_it->first);
150  if (parsed_it == decoded_keys_.end()) {
151  std::cerr << "Warning: Found non-parsed key " << map_it->first
152  << " in MsgPack MAP.\n";
153  }
154  }
155 }
